Actor Demon64 3002
{
	Game Doom64
	Health 150
	Radius 44
	Height 100
	Mass 400
	PainChance 180
	Speed 12
	MeleeRange 60
	ActiveSound "demon/active"
	AttackSound "demon/melee"
	DeathSound "demon/death"
	PainSound "demon/pain"
	SeeSound "demon/sight"
	Monster
	States
	{
	Spawn:
		SARG BD 8 A_Look
		Loop
	See:
		SARG AABBCCDD 2 A_Chase
		Loop
	Melee:
		SARG EF 8 A_FaceTarget
		SARG G 8 A_CustomMeleeAttack(4*random(1,8), "demon/melee")
		Goto See
	Pain:
		SARG H 2
		SARG H 2 A_Pain
		Goto See
	Death:
		SARG I 8
		SARG J 8 A_Scream
		SARG K 4
		SARG L 4 A_Fall
		SARG M 4 //A_OnDeathTrigger
		SARG N -1
		Stop
	}
}
