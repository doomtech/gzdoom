// Summoning Doll -----------------------------------------------------------

ACTOR SummoningDoll
{
	Info
	{
		Game ("Hexen")
	}

	DefaultProperties
	{
		Speed (20)
		+NOBLOCKMAP +DROPOFF +MISSILE
		+NOTELEPORT
	}

	native action A_Summon();

	States
	{
	Spawn:
		SUMN A 4;
		Loop
	Death:
		SUMN AA 4;
		SUMN A 4 A_Summon;
		Stop
	}
}
